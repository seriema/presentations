Tree.prototype.printChildren = function() {
  this.children.forEach(function(child) {
    console.log('Child ' + child.value, 'Parent ' + this.value); // wrong this
  };
};

Tree.prototype.printChildren = function() {
  var self = this;
  this.children.forEach(function(child) {
    console.log('Child ' + child.value, 'Parent ' + self.value);
  };
};

Safer constructors


function User(name, passwordHash) {
  if (!(this instanceof User))
    return new User(name, passwordHash);

  this.name = name;
  this.passwordHash = passwordHash;
}

var user1 = new User('abc', '123'); // as expected
var user2 = User('def', '456'); // now also works!

Even safer constructor


function User(name, passwordHash) {
  var self = this instanceof User ? this : Object.create(User.prototype);

  self.name = name;
  self.passwordHash = passwordHash;

  return self;
}

var user1 = new User('abc', '123'); // as expected
var user2 = User('def', '456'); // now also works!
